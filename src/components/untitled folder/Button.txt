---
// Premium Button Component - Sophisticated interaction design
interface Props {
  href?: string;
  variant?: 'primary' | 'secondary' | 'outline' | 'text';
  size?: 'sm' | 'md' | 'lg';
  class?: string;
  target?: string;
  rel?: string;
  type?: 'button' | 'submit' | 'reset';
}

const { 
  href, 
  variant = 'primary', 
  size = 'md',
  class: className = '',
  target,
  rel,
  type = 'button'
} = Astro.props;

// Sophisticated base classes with premium interactions
const baseClasses = "relative inline-flex items-center justify-center font-semibold tracking-tight transition-all duration-300 ease-out focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-accent-primary focus-visible:ring-offset-2 focus-visible:ring-offset-surface-primary disabled:opacity-50 disabled:cursor-not-allowed overflow-hidden";

// Mathematical sizing using 8pt grid
const sizeClasses = {
  sm: "h-10 px-4 text-sm rounded-lg gap-2", // 40px height
  md: "h-12 px-6 text-base rounded-lg gap-2", // 48px height
  lg: "h-14 px-8 text-lg rounded-xl gap-3", // 56px height
};

// Premium variant treatments
const variantClasses = {
  // Primary: Sophisticated gradient with glow effects
  primary: "bg-gradient-to-r from-accent-primary via-accent-active to-accent-primary text-white shadow-lg shadow-accent-primary/25 hover:shadow-xl hover:shadow-accent-primary/40 hover:-translate-y-0.5 active:translate-y-0 bg-size-200 hover:bg-right",
  
  // Secondary: Elegant border with sophisticated hover
  secondary: "border-2 border-accent-primary/60 text-accent-primary bg-surface-primary/80 backdrop-blur-sm hover:border-accent-primary hover:bg-accent-primary/10 hover:shadow-lg hover:shadow-accent-primary/20",
  
  // Outline: Clean and minimal
  outline: "border border-content-tertiary text-content-primary hover:border-accent-primary hover:text-accent-primary hover:bg-accent-primary/5",
  
  // Text: Sophisticated underline animation
  text: "text-accent-primary hover:text-accent-secondary relative before:absolute before:bottom-0 before:left-0 before:w-0 before:h-0.5 before:bg-accent-primary before:transition-all before:duration-300 hover:before:w-full",
};

const Tag = href ? 'a' : 'button';
---

<Tag 
  href={href}
  target={target}
  rel={rel}
  type={!href ? type : undefined}
  class:list={[
    baseClasses, 
    sizeClasses[size],
    variantClasses[variant], 
    className
  ]}
>
  <!-- Sophisticated glow effect for primary variant -->
  {variant === 'primary' && (
    <div class="absolute inset-0 bg-gradient-to-r from-accent-primary to-accent-active rounded-lg blur-sm opacity-0 group-hover:opacity-30 transition-opacity duration-300 -z-10"></div>
  )}
  
  <!-- Subtle overlay for premium feel -->
  <div class="absolute inset-0 bg-gradient-to-t from-black/5 to-transparent rounded-lg"></div>
  
  <!-- Content with proper z-index -->
  <span class="relative z-10 flex items-center justify-center">
    <slot />
  </span>
</Tag>
